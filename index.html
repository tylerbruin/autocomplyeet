Setup

            autocomplyeet.options = {
                id: "#search-text-input",   // QuerySelector of your search input element (required)
                cloneCSS: true,             // True|false - clones entire Computed styles of Input element (required)
                cache: 15,                  // Amount of ajax calls to cache. (required)
                                            // URL to Funnelback server, with partial_query at end of string. (required)
                server: "https://{Server Path}/s/suggest.json?collection={Collection Name}&show=10&partial_query=" 
            };
            autocomplyeet.fn.init();
